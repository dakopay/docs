## get Subscription

```jsx
// Use
import { subscriptions } from 'repa-sdk';

var data = subscriptions(_hash);
```

## get random Subscription

```jsx
// Use
import { randomSubscription } from 'repa-sdk';

var data = randomSubscription();
```

## get Subscription by user

```jsx
// Use
import { getSubscriptionsByUser } from 'repa-sdk';

var data = getSubscriptionsByUser(_user);
```
